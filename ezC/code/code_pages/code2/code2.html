2. Program to Calculate area of a Triangle.

                
#include<stdio.h>
#include<math.h>

int main()
{
    float a,b,c,s,area;
                    
    clrscr();
                    
    printf("Enter the sides of the triangle : ");
    scanf("%f%f%f",&a,&b,&c);
    s = (a+b+c)/2;
    area = sqrt( s*(s-a)*(s-b)*(s-c) );
    printf("The area of the triangle is %0.2f", area);
                    
    getch();
    return 0;
}
